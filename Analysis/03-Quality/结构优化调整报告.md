### 3.2 命名规范统一

#### 3.2.1 文件命名规范

**规范标准**:

```python
# 文件命名规范
class FileNamingConvention:
    def __init__(self):
        self.convention = {
            "format": "NN-Topic.md",           # 数字-主题.md
            "language": "chinese",              # 中文命名
            "separator": "-",                   # 连字符分隔
            "max_length": 50,                   # 最大长度50字符
            "forbidden_chars": ["\\", "/", ":", "*", "?", "\"", "<", ">", "|"]
        }
    
    def standardize_filename(self, filename: str) -> str:
        """标准化文件名"""
        # 1. 移除特殊字符
        cleaned = self.remove_forbidden_chars(filename)
        
        # 2. 转换为中文
        chinese = self.translate_to_chinese(cleaned)
        
        # 3. 添加编号前缀
        numbered = self.add_number_prefix(chinese)
        
        # 4. 限制长度
        truncated = self.truncate_if_needed(numbered)
        
        return truncated
```

**实施效果**:

- 统一了文件命名格式
- 建立了编号系统
- 提高了文件的可读性

```rust
// 主题逻辑分组
#[derive(Debug, Clone)]
struct TopicLogicalGrouping {
    primary_topics: Vec<String>,
    secondary_topics: Vec<String>,
    cross_references: HashMap<String, Vec<String>>,
}

impl TopicLogicalGrouping {
    fn group_by_logic(&self, files: &[File]) -> GroupedFiles {
        let mut grouped = GroupedFiles::new();
        
        for file in files {
            // 1. 识别主题
            let topic = self.identify_topic(file);
            
            // 2. 确定分组
            let group = self.determine_group(topic);
            
            // 3. 建立关系
            let relationships = self.establish_relationships(file, group);
            
            grouped.add_file(file, group, relationships);
        }
        
        grouped
    }
}
```

**实施效果**:

- 按主题逻辑重新分组
